<script>
// 这里替换成你的 Worker 真实地址
const WORKER_URL = 'https://vocaloid-api.925857461.workers.dev/';

async function fetchRandomSong() {
    const card = document.getElementById('vc-card');
    const btn = document.getElementById('vc-refresh-btn');
    const title = document.getElementById('vc-title');
    const pname = document.getElementById('vc-pname');
    const singer = document.getElementById('vc-singer');
    const link = document.getElementById('vc-bili-link');

    card.classList.add('loading');
    btn.disabled = true;

    try {
        const response = await fetch(WORKER_URL);
        const data = await response.json();

        if (data.error) throw new Error(data.error);

        // 渲染数据
        title.innerText = data.title;
        pname.innerText = "P主: " + data.p_name;
        singer.innerText = data.singer;
        
        // 更新跳转链接
        link.href = data.link;
        link.style.display = 'block';

    } catch (err) {
        console.error(err);
        title.innerText = "信号同步失败";
        pname.innerText = "请检查 Worker 配置或跨域设置";
    } finally {
        card.classList.remove('loading');
        btn.disabled = false;
    }
}

// 页面加载时自动运行一次
document.addEventListener('DOMContentLoaded', fetchRandomSong);
</script>
